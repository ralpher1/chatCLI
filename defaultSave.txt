```
  { id: 5, name: 'Product 5', price: 30 },::::
];

const ProductList = () => {
  const [cart, setCart] = useState([]);

  const addToCart = (product) => {
    setCart([...cart, product]);
  };

  const productsComponents = products.map((product) => {
    return (
      <Product
        key={product.id}
        product={product}
        addToCart={addToCart}
      />
    );
  });

  return (
    <div>
      <ul>{productsComponents}</ul>
      <p>Cart: {cart.length} items</p>::::
    </div>
  );
}

export default ProductList;
```

```
import React from 'react';
```


This component is named `MyComponent`, imports the `React` library, takes a prop called `greeting`, and returns a `div` containing an `h1` tag. When rendered, it will display the passed greeting prop in an `h1` tag.

Here is an example of how it can be used in another file:

```// App.js
import React from 'react';
import MyComponent from './MyComponent';

const App = () => (
  <div>
    <MyComponent greeting="Hello, World!" />
  </div>
);

export default App;
```

This file imports the `React` and `MyComponent` libraries, and renders the `MyComponent` component while passing a prop `greeting` with a value of `Hello, World!`. When rendered, it will display the text "Hello, World!" in an `h1` tag.
